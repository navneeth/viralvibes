751def create_tabs( 

752 tabs: List[Tuple[str, "Component", Optional[str]]], 

753 tabs_id: str, 

754 alt: bool = True, 

755 tab_cls: str = "uk-active", 

756 container_cls: str = "space-y-4", 

757) -> Div: 

758 """Creates a MonsterUI tab component with optional icons and improved styling.""" 

759 tab_links = [] 

760 tab_content = [] 

761 

762 for i, tab in enumerate(tabs): 

763 if len(tab) == 3: 

764 title, content, icon = tab 

765 label = Div( 

766 UkIcon(icon, cls="mr-2 text-lg") if icon else None, 

767 Span(title), 

768 cls="flex items-center", 

769 ) 

770 else: 

771 title, content = tab[:2] 

772 label = Span(title) 

773 

774 # Active class applied to <li>, not <a> 

775 li_cls = tab_cls if i == 0 else "" 

776 tab_links.append(Li(A(label, href="#"), cls=li_cls)) 

777 

778 # Guard against None content 

779 tab_content.append(Li(Div(content, cls="p-2"))) 

780 

781 return Container( 

782 TabContainer( 

783 *tab_links, 

784 uk_switcher=f"connect: #{tabs_id}; animation: uk-animation-fade", 

785 alt=alt, 

786 cls="mb-4 flex gap-2", 

787 ), 

788 Ul(id=tabs_id, cls="uk-switcher")(*tab_content), 

789 cls=container_cls, 

790 )
