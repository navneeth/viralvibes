56def _StatusBar(label: str, count: int, total: int, color_cls: str) -> Div: 

57 pct = round(count / total * 100, 1) if total else 0.0 

58 bar_w = max(int(pct), 2) if count > 0 else 0 

59 return Div( 

60 Span(label, cls="text-sm font-medium text-foreground w-20 inline-block"), 

61 Div( 

62 Div(cls=f"{color_cls} h-3 rounded-full", style=f"width:{bar_w}%"), 

63 cls="flex-1 bg-accent rounded-full mx-3 h-3", 

64 ), 

65 Span(f"{count:,}", cls="text-sm font-mono text-muted-foreground w-20 text-right"), 

66 Span(f"{pct}%", cls="text-sm font-mono text-muted-foreground w-14 text-right"), 

67 cls="flex items-center py-1", 

68 )

71def _JobRow(job: dict) -> Tr: 

72 status = job.get("status", "unknown") 

73 status_cls = _JOB_STATUS_CLS.get(status, "text-muted-foreground bg-accent") 

74 icon = "✅" if status in ("completed", "synced") else "❌" if status == "failed" else "⏳" 

75 

76 started = job.get("started_at") or job.get("created_at") 

77 finished = job.get("finished_at") or job.get("updated_at") 

78 duration = "—" 

79 if started and finished: 

80 try: 

81 s = datetime.fromisoformat(started.replace("Z", "+00:00")) 

82 f = datetime.fromisoformat(finished.replace("Z", "+00:00")) 

83 secs = int((f - s).total_seconds()) 

84 duration = f"{secs}s" if secs >= 0 else "—" 

85 except Exception: 

86 pass 

87 

88 creator_id = str(job.get("creator_id") or "")[:8] 

89 return Tr( 

90 Td(Code(f"#{job.get('id', '')}", cls="text-xs"), cls="px-3 py-2"), 

91 Td(Code(creator_id + "…", cls="text-xs text-primary"), cls="px-3 py-2"), 

92 Td( 

93 Span(status, cls=f"text-xs font-medium px-2 py-0.5 rounded-full {status_cls}"), 

94 cls="px-3 py-2", 

95 ), 

96 Td(duration, cls="px-3 py-2 text-xs font-mono text-muted-foreground"), 

97 Td( 

98 Span(f"retry:{job.get('retry_count', 0)}", cls="text-xs text-muted-foreground"), 

99 cls="px-3 py-2", 

100 ), 

101 Td(icon, cls="px-3 py-2 text-center"), 

102 cls="border-b border-border hover:bg-accent/50", 

103 )

137def _ThroughputSection(throughput: dict) -> Div: 

138 def _row(label, value): 

139 return DivFullySpaced( 

140 Span(label, cls="text-sm text-muted-foreground"), 

141 Span(value, cls="text-sm font-mono font-medium text-foreground"), 

142 cls="py-1.5 border-b border-border last:border-0", 

143 ) 

144 

145 quota_used = throughput.get("quota_used", 0) 

146 quota_total = throughput.get("quota_total", 10000) 

147 quota_pct = round(quota_used / quota_total * 100, 1) if quota_total else 0 

148 return Card( 

149 H3( 

150 "Worker Health", 

151 cls="text-sm font-mono uppercase tracking-widest text-muted-foreground mb-4", 

152 ), 

153 _row("Jobs / hour (est.)", str(throughput.get("jobs_per_hour", "—"))), 

154 _row("Avg duration", throughput.get("avg_duration", "—")), 

155 _row("Last job", throughput.get("last_job_ago", "—")), 

156 _row("YT Quota", f"{quota_used:,} / {quota_total:,} ({quota_pct}%)"), 

157 Div( 

158 Div( 

159 cls="bg-orange-400 h-2 rounded-full transition-all", 

160 style=f"width:{min(quota_pct, 100)}%", 

161 ), 

162 cls="w-full bg-accent rounded-full h-2 mt-3", 

163 ), 

164 body_cls="p-5", 

165 )
