17def StepProgress( 

18 completed_steps: int = 0, 

19 steps_config: List[Tuple[str, str, str]] = PLAYLIST_STEPS_CONFIG, 

20 steps_cls: str = STEPS_CLS, 

21) -> Steps: 

22 """Create a step progress component. 

23 

24 Args: 

25 completed_steps: Number of completed steps 

26 steps_config: List of (title, icon, description) tuples 

27 steps_cls: CSS classes for the Steps container 

28 Returns: 

29 Steps component with styled progress indicators 

30 """ 

31 steps = [] 

32 for i, (title, icon, description) in enumerate(steps_config): 

33 # determine step state 

34 if i < completed_steps: 

35 step_cls = StepT.success 

36 elif i == completed_steps: 

37 step_cls = StepT.primary 

38 else: 

39 step_cls = StepT.neutral 

40 

41 steps.append( 

42 LiStep(title, cls=step_cls, data_content=icon, description=description) 

43 ) 

44 

45 return Steps(*steps, cls=steps_cls)
