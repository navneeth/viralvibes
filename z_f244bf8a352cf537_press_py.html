37def _safe_hero_stats() -> dict: 

38 """Return creator hero stats, or a conservative fallback on any failure.""" 

39 try: 

40 # Imported lazily so a Supabase import-time error can't kill the page. 

41 from db import get_creator_hero_stats 

42 

43 return get_creator_hero_stats() or {} 

44 except Exception: 

45 logger.warning("press: get_creator_hero_stats failed", exc_info=True) 

46 return {}

65def _format_count(n: int | float | None) -> str: 

66 """Press-page count formatter — ``format_number`` with an em-dash for missing data. 

67 

68 Wraps the shared ``utils.formatting.format_number`` so the press page 

69 matches numbers shown elsewhere (e.g. nav, hero) instead of inventing 

70 its own "807K" vs "807.0K" rounding rules. Only ``None`` is treated as 

71 missing — a legitimate zero (e.g. "0 creators in a niche") passes 

72 through to ``format_number`` instead of being silently hidden as "—". 

73 """ 

74 if n is None: 

75 return "—" 

76 return format_number(n)
