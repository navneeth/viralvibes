1""" 

2Outreach helpers for saved creators. 

3 

4The first version deliberately exports contact-ready rows instead of sending

45def _channel_url(creator: dict[str, Any]) -> str: 

46 channel_id = creator.get("channel_id") or "" 

47 return creator.get("channel_url") or ( 

48 f"https://www.youtube.com/channel/{channel_id}" if channel_id else "" 

49 )

52def outreach_angle(creator: dict[str, Any]) -> str: 

53 """Short note useful for manual personalization in an external tool.""" 

54 name = creator.get("channel_name") or "This creator" 

55 category = creator.get("primary_category") or creator.get("category") or "their niche" 

56 grade = creator.get("quality_grade") or "" 

57 subs_delta = creator.get("subscribers_change_30d") 

58 

59 growth = "" 

60 if isinstance(subs_delta, int) and subs_delta > 0: 

61 growth = f" and gained {subs_delta:,} subscribers in the last 30 days" 

62 

63 grade_note = f" with a {grade} quality grade" if grade else "" 

64 return f"{name} is active in {category}{grade_note}{growth}."

67def creator_to_outreach_row( 

68 creator: dict[str, Any], *, base_url: str = "https://www.viralvibes.fyi" 

69) -> dict[str, str]: 

70 """ 

71 Convert a creator row into a CSV import row. 

72 

73 The row intentionally uses broadly recognized import column names. Tools 

74 that require custom mapping should still auto-detect Email, Company, 

75 Website, Notes, and Tags cleanly. 

76 """ 

77 contacts = extract_contact_signals_from_creator(creator) 

78 creator_id = str(creator.get("id") or "") 

79 category = creator.get("primary_category") or creator.get("category") or "" 

80 country = creator.get("country_code") or "" 

81 language = creator.get("default_language") or creator.get("language") or "" 

82 

83 tags = [tag for tag in ("viralvibes", "saved-creator", category, country) if tag] 

84 

85 return { 

86 "Email": contacts.email, 

87 "First Name": "", 

88 "Last Name": "", 

89 "Company": str(creator.get("channel_name") or ""), 

90 "Website": contacts.website_url, 

91 "YouTube URL": _channel_url(creator), 

92 "Instagram URL": contacts.instagram_url, 

93 "X URL": contacts.x_url, 

94 "TikTok URL": contacts.tiktok_url, 

95 "LinkedIn URL": contacts.linkedin_url, 

96 "Tags": ", ".join(tags), 

97 "Notes": outreach_angle(creator), 

98 "Subscribers": str(creator.get("current_subscribers") or 0), 

99 "Views": str(creator.get("current_view_count") or 0), 

100 "Videos": str(creator.get("current_video_count") or 0), 

101 "Quality Grade": str(creator.get("quality_grade") or ""), 

102 "Engagement Score": str(creator.get("engagement_score") or ""), 

103 "30 Day Subscriber Growth": str(creator.get("subscribers_change_30d") or ""), 

104 "30 Day View Growth": str(creator.get("views_change_30d") or ""), 

105 "Category": str(category), 

106 "Country": str(country), 

107 "Language": str(language), 

108 "ViralVibes Profile URL": ( 

109 f"{base_url.rstrip('/')}/creator/{creator_id}" if creator_id else "" 

110 ), 

111 }

114def build_outreach_rows( 

115 creators: list[dict[str, Any]], *, base_url: str = "https://www.viralvibes.fyi" 

116) -> list[dict[str, str]]: 

117 return [creator_to_outreach_row(c, base_url=base_url) for c in creators]

120def filter_email_ready_rows(rows: list[dict[str, str]]) -> list[dict[str, str]]: 

121 """Rows suitable for email tools, which usually require a non-empty email.""" 

122 return [row for row in rows if row.get("Email")]

125def render_outreach_csv(rows: list[dict[str, str]]) -> str: 

126 buf = io.StringIO() 

127 writer = csv.DictWriter(buf, fieldnames=EMAIL_EXPORT_HEADERS, extrasaction="ignore") 

128 writer.writeheader() 

129 writer.writerows(rows) 

130 return buf.getvalue()
