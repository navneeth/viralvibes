1"""Editors' Shortlist rail — curated entry point into ``/creators/top``. 

2 

3A horizontal card rail surfacing the five A+ category landing pages plus 

4the global A+ shortlist. Designed to live near the top of the two main 

5discovery surfaces (``/creators`` and ``/lists``) so anonymous visitors 

6and returning users alike have a one-click path into the editorial cut. 

7 

8Design notes 

9------------ 

10* Single accent treatment (no per-card colour) — restraint over noise. 

11* Mono numerals for counts so they read as data, not marketing. 

12* Horizontal scroll on mobile via ``overflow-x-auto`` so the rail never 

13 reflows into a tall column on small screens. 

14* Counts are optional. When omitted the cards still render with a 

15 generic CTA, which lets pages that don't want to take the small count 

16 query opt out cleanly. 

17""" 

18 

19from __future__ import annotations 

20 

21from fasthtml.common import A, Div, H3, P, Span 

22 

23from utils import format_number 

24 

25__all__ = ["EditorsShortlistRail"]

27# The "All A+" card is always first as the broadest entry point. The 

28# category cards are derived from ``routes.creators.TOP_CATEGORY_SLUGS`` 

29# at call time (see ``_build_rail_items``) so the rail and the 

30# ``/creators/top`` landing pages can never silently drift apart. The 

31# import is deferred to avoid a components → routes → views → components 

32# import cycle at module load. 

33_ALL_AP_ITEM: tuple[None, str, str] = (None, "All A+ Tier", "/creators/top") 

34

36def _build_rail_items() -> list[tuple[str | None, str, str]]: 

37 """Return ``[(slug_or_None, label, href), ...]`` for the rail. 

38 

39 Order: the synthetic "All A+" card first, then category cards in the 

40 order ``TOP_CATEGORY_SLUGS`` defines them (which we keep sorted by A+ 

41 population at the source). 

42 """ 

43 from routes.creators import TOP_CATEGORY_SLUGS # local: avoid circular import 

44 

45 items: list[tuple[str | None, str, str]] = [_ALL_AP_ITEM] 

46 items.extend( 

47 (slug, label, f"/creators/top/{slug}") for slug, label in TOP_CATEGORY_SLUGS.items() 

48 ) 

49 return items

52def _rail_card( 

53 *, 

54 label: str, 

55 href: str, 

56 count: int | None, 

57 is_all: bool, 

58) -> A: 

59 """Single card in the rail. Anchor element so the entire surface is clickable.""" 

60 count_block = ( 

61 Div( 

62 Span( 

63 format_number(count), 

64 cls="block text-3xl font-mono font-semibold text-foreground tabular-nums", 

65 ), 

66 P( 

67 "A+ rated creators", 

68 cls="text-xs text-muted-foreground mt-1", 

69 ), 

70 cls="mt-4", 

71 ) 

72 if count is not None 

73 else Div( 

74 P( 

75 "View top creators", 

76 cls="text-sm text-foreground/80 mt-4", 

77 ), 

78 ) 

79 ) 

80 

81 eyebrow = "Editors' Pick" if is_all else "Category" 

82 

83 return A( 

84 Div( 

85 Span( 

86 eyebrow.upper(), 

87 cls=("text-[10px] font-mono tracking-[0.2em] " "text-muted-foreground/70"), 

88 ), 

89 H3( 

90 label, 

91 cls="text-lg font-semibold text-foreground mt-2 leading-tight", 

92 ), 

93 count_block, 

94 Div( 

95 Span( 

96 "Browse →", 

97 cls=( 

98 "text-xs font-medium text-muted-foreground " 

99 "group-hover:text-primary transition-colors" 

100 ), 

101 ), 

102 cls="mt-6 pt-4 border-t border-border/60", 

103 ), 

104 cls="h-full flex flex-col", 

105 ), 

106 href=href, 

107 cls=( 

108 "group block snap-start shrink-0 " 

109 "w-[220px] sm:w-[240px] " 

110 "rounded-xl border border-border bg-card " 

111 "p-5 no-underline " 

112 "transition-all duration-200 " 

113 "hover:border-primary/40 hover:bg-primary/[0.03] " 

114 "hover:-translate-y-0.5 hover:shadow-sm" 

115 ), 

116 )

119def EditorsShortlistRail( 

120 counts: dict[str, int] | None = None, 

121 *, 

122 headline: str = "Editors' Shortlist", 

123 subhead: str | None = "The A+ engagement tier — hand-graded, ranked by reach.", 

124 see_all_label: str = "See all A+ creators →", 

125 see_all_href: str = "/creators/top", 

126) -> Div: 

127 """Render the rail. Pass ``counts`` from ``get_aplus_category_counts()``. 

128 

129 The rail intentionally returns a plain ``Div`` (no Container) so callers 

130 control the surrounding padding and section rhythm. 

131 """ 

132 if counts is None: 

133 counts = {} 

134 

135 cards = [ 

136 _rail_card( 

137 label=label, 

138 href=href, 

139 count=counts.get(slug or "all"), 

140 is_all=slug is None, 

141 ) 

142 for slug, label, href in _build_rail_items() 

143 ] 

144 

145 return Div( 

146 # Header strip — H2-equivalent + optional see-all CTA on one row. 

147 Div( 

148 Div( 

149 Span( 

150 "EDITORIAL", 

151 cls=( 

152 "block text-[10px] font-mono tracking-[0.22em] " 

153 "text-muted-foreground/70 mb-2" 

154 ), 

155 ), 

156 H3( 

157 headline, 

158 cls="text-xl sm:text-2xl font-bold text-foreground tracking-tight", 

159 ), 

160 (P(subhead, cls="text-sm text-muted-foreground mt-1") if subhead else None), 

161 ), 

162 A( 

163 see_all_label, 

164 href=see_all_href, 

165 cls=( 

166 "hidden sm:inline-flex items-center text-sm font-medium " 

167 "text-muted-foreground hover:text-primary transition-colors " 

168 "no-underline whitespace-nowrap" 

169 ), 

170 ), 

171 cls="flex items-end justify-between gap-4 mb-5", 

172 ), 

173 # Rail itself — horizontal scroll with snap. 

174 Div( 

175 *cards, 

176 cls=( 

177 "flex gap-4 overflow-x-auto snap-x snap-mandatory " 

178 "-mx-4 px-4 sm:mx-0 sm:px-0 pb-2" 

179 ), 

180 ), 

181 cls="my-10 sm:my-12", 

182 )
