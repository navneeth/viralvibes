29def _section(heading: str, *paragraphs) -> Div: 

30 return Div( 

31 *([H2(heading, cls="text-xl font-semibold text-foreground mb-3")] if heading else []), 

32 *[P(text, cls="text-muted-foreground leading-relaxed mb-3") for text in paragraphs],

36def _bullet_section(heading: str, items: list[str]) -> Div: 

37 return Div( 

38 *([H2(heading, cls="text-xl font-semibold text-foreground mb-3")] if heading else []), 

39 Ul( 

40 *[Li(item, cls="text-muted-foreground leading-relaxed mb-2") for item in items], 

41 cls="list-disc list-inside space-y-1", 

42 ), 

43 )
