41def _pct(part: float, total: float) -> float: 

42 """Fraction of total as 0–100, clamped.""" 

43 if not total: 

44 return 50.0 

45 return max(4.0, min(96.0, 100.0 * part / total))

54def _ratio_bar( 

55 val_a: float, val_b: float, colour_a: str = "bg-blue-500", colour_b: str = "bg-violet-500" 

56): 

57 """ 

58 Single proportional bar: A fills from left, B from right. 

59 Width is proportional to share of (A + B). 

60 """ 

61 total = val_a + val_b 

62 pct_a = _pct(val_a, total) 

63 pct_b = 100.0 - pct_a 

64 return Div( 

65 Div(cls=f"h-2.5 {colour_a} rounded-l-full", style=f"width:{pct_a:.1f}%"), 

66 Div(cls=f"h-2.5 {colour_b} rounded-r-full", style=f"width:{pct_b:.1f}%"), 

67 cls="flex w-full h-2.5 bg-border rounded-full overflow-hidden my-2", 

68 )

109def _metric_row( 

110 label: str, 

111 val_a, 

112 val_b, 

113 *, 

114 higher_is_better: bool = True, 

115 unit: str = "", 

116 val_a_raw: float | None = None, 

117 val_b_raw: float | None = None, 

118 show_bar: bool = False, 

119 colour_a: str = "bg-blue-500", 

120 colour_b: str = "bg-violet-500", 

121): 

122 """ 

123 A single comparison row: 

124 label | val_a 🏆? | bar (optional) | 🏆? val_b 

125 """ 

126 raw_a = ( 

127 val_a_raw 

128 if val_a_raw is not None 

129 else (float(str(val_a).replace(",", "").replace("%", "") or 0) if val_a else 0.0) 

130 ) 

131 raw_b = ( 

132 val_b_raw 

133 if val_b_raw is not None 

134 else (float(str(val_b).replace(",", "").replace("%", "") or 0) if val_b else 0.0) 

135 ) 

136 

137 if raw_a == raw_b: 

138 win_a = win_b = False 

139 elif higher_is_better: 

140 win_a, win_b = raw_a > raw_b, raw_b > raw_a 

141 else: 

142 win_a, win_b = raw_a < raw_b, raw_b < raw_a 

143 

144 trophy_a = _trophy(win_a) 

145 trophy_b = _trophy(win_b) 

146 

147 disp_a = Div( 

148 Span( 

149 str(val_a) + unit if val_a is not None else "—", 

150 cls="text-sm font-semibold text-foreground", 

151 ), 

152 trophy_a, 

153 cls="flex items-center gap-1 justify-end", 

154 ) 

155 disp_b = Div( 

156 trophy_b, 

157 Span( 

158 str(val_b) + unit if val_b is not None else "—", 

159 cls="text-sm font-semibold text-foreground", 

160 ), 

161 cls="flex items-center gap-1", 

162 ) 

163 

164 bar = _ratio_bar(raw_a or 0.001, raw_b or 0.001, colour_a, colour_b) if show_bar else None 

165 

166 return Div( 

167 # 3-column: left value | bar | right value 

168 # On mobile the bar collapses to a line between the two values 

169 Div( 

170 Span(label, cls="text-xs text-muted-foreground col-span-3 sm:hidden block mb-1"), 

171 Div(disp_a, cls="flex-1 text-right"), 

172 Div( 

173 Span(label, cls="text-xs text-muted-foreground text-center hidden sm:block"), 

174 bar, 

175 cls="w-32 sm:w-40 flex flex-col items-center justify-center px-2 shrink-0", 

176 ), 

177 Div(disp_b, cls="flex-1"), 

178 cls="flex items-center gap-2", 

179 ), 

180 cls="py-2 border-b border-border last:border-0", 

181 )

679 def _score_creator(name, subs, growth_rate, eng, vps, uploads, grade): 

680 """Simple weighted score — who 'wins' overall.""" 

681 score = 0 

682 score += 2 if subs > 0 else 0 # presence 

683 score += 3 if growth_rate > 1 else (1 if growth_rate > 0 else 0) 

684 score += 3 if eng > 7 else (1 if eng > 4 else 0) 

685 score += 2 if vps > 200 else (1 if vps > 50 else 0) 

686 score += 1 if uploads > 4 else 0 

687 score += {"A+": 3, "A": 2, "B+": 1}.get(grade, 0) 

688 return score 

689 

690 score_a = _score_creator(name_a, subs_a, growth_rate_a, eng_a, vps_a, uploads_a, grade_a) 

691 score_b = _score_creator(name_b, subs_b, growth_rate_b, eng_b, vps_b, uploads_b, grade_b)
