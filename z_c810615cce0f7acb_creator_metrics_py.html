42def estimate_monthly_revenue_v4( 

43 total_subs: int, 

44 total_views: int, 

45 video_count: int, 

46 country_code: str = "US", 

47 niche: str = "general", 

48) -> dict: 

49 """ 

50 2026 model: tuned for the Shorts-heavy YouTube economy. 

51 

52 All inputs come directly from the YouTube API; no extra worker quota needed. 

53 

54 Args: 

55 total_subs: statistics.subscriberCount 

56 total_views: statistics.viewCount (lifetime) 

57 video_count: statistics.videoCount 

58 country_code: snippet.country (2-letter ISO, e.g. "US") 

59 niche: primary_category from DB (e.g. "Gaming", "Finance") 

60 

61 Returns a dict with keys: 

62 est_monthly_total float — total estimated monthly revenue ($) 

63 revenue_split dict — breakdown: adsense_long, adsense_shorts, brand_deals 

64 assumed_shorts_pct str — e.g. "80%" 

65 """ 

66 rates = _MARKET_DATA.get(country_code.upper(), _MARKET_DATA["DEFAULT"]) 

67 

68 # View-mix heuristic: high video_count relative to subs → Shorts factory 

69 shorts_ratio = 0.8 if (video_count / (total_subs + 1)) > 0.01 else 0.4 

70 

71 # Estimate monthly views (≈6% of lifetime total for an active channel) 

72 monthly_views = total_views * 0.06 

73 views_long = monthly_views * (1 - shorts_ratio) 

74 views_shorts = monthly_views * shorts_ratio 

75 

76 # Niche multiplier — case-insensitive lookup, default 1.0 

77 niche_mod = _NICHE_MULTIPLIERS.get(niche.lower(), 1.0) 

78 

79 # Ad revenue (AdSense) 

80 ad_rev = (views_long / 1_000) * rates["long"] * niche_mod + (views_shorts / 1_000) * rates[ 

81 "shorts" 

82 ] 

83 

84 # Sponsorships: $25 CPM on long-form views only (brands ignore Shorts) 

85 sponsorships = (views_long / 1_000) * 25.0 * niche_mod 

86 

87 # Direct monetisation (merch/memberships): scales with loyal long-form audience 

88 direct_mon = ad_rev * (0.25 if total_subs > 500_000 else 0.05) 

89 

90 total_monthly = ad_rev + sponsorships + direct_mon 

91 

92 return { 

93 "est_monthly_total": round(total_monthly, 2), 

94 "revenue_split": { 

95 "adsense_long": round((views_long / 1_000) * rates["long"] * niche_mod, 2), 

96 "adsense_shorts": round((views_shorts / 1_000) * rates["shorts"], 2), 

97 "brand_deals": round(sponsorships, 2), 

98 }, 

99 "assumed_shorts_pct": f"{int(shorts_ratio * 100)}%", 

100 }

108def calculate_momentum_score( 

109 views_change_30d: int | None, 

110 subs_change_30d: int | None, 

111 current_subs: int, 

112) -> float | None: 

113 """ 

114 Compute a 0–100 momentum score that combines subscriber growth velocity 

115 with view velocity. 

116 

117 The score blends two signals with equal weight: 

118 - Sub velocity: subs_change_30d / current_subs * 100 (30-day growth %) 

119 - View velocity: views_change_30d / current_subs (viral coefficient) 

120 

121 Both are normalised against representative ceilings so the result sits 

122 naturally in the 0–100 range for typical creators: 

123 - sub_pct normalised at 5 % (5 % 30d growth → score contribution 50) 

124 - viral_coeff normalised at 5.0 (viral_coeff 5 → score contribution 50) 

125 

126 Returns None when neither delta is available (channel not yet tracked). 

127 """ 

128 if views_change_30d is None and subs_change_30d is None: 

129 return None 

130 if current_subs <= 0: 

131 return None 

132 

133 _views = max(views_change_30d or 0, 0) 

134 _subs = max(subs_change_30d or 0, 0) 

135 

136 # Normalised sub growth (ceiling = 5%) 

137 sub_component = min((_subs / current_subs * 100) / 5.0, 1.0) * 50 

138 

139 # Normalised view velocity / viral coeff (ceiling = 5.0) 

140 view_component = min((_views / current_subs) / 5.0, 1.0) * 50 

141 

142 return round(sub_component + view_component, 1)

410def get_language_name(language_code: str) -> str: 

411 """ 

412 Return the English display name for a base ISO 639-1 language code. 

413 

414 Lookup order: 

415 1. Curated ``_LANGUAGE_NAMES`` dict (fast, covers all YouTube-common codes). 

416 2. ``pycountry.languages`` by alpha_2 (ISO 639-1 two-letter codes). 

417 3. ``pycountry.languages`` by alpha_3 (ISO 639-2/3 three-letter codes, 

418 e.g. ``fil`` for Filipino). 

419 4. Uppercase code as last resort (e.g. ``"XYZ"``). 

420 

421 The ISO name is cleaned by stripping any parenthetical qualifier 

422 (e.g. ``"Malay (individual language)"`` → ``"Malay"`` ). 

423 """ 

424 code = language_code.lower() 

425 if code in _LANGUAGE_NAMES: 

426 return _LANGUAGE_NAMES[code] 

427 

428 # pycountry fallback — try alpha_2 first, then alpha_3 for longer codes 

429 lang = pycountry.languages.get(alpha_2=code) or pycountry.languages.get(alpha_3=code) 

430 if lang: 

431 # Strip verbose ISO parentheticals like " (macrolanguage)" or " (individual language)" 

432 return lang.name.split(" (")[0] 

433 

434 return code.upper()
