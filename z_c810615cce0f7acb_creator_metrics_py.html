41def estimate_monthly_revenue_v4( 

42 total_subs: int, 

43 total_views: int, 

44 video_count: int, 

45 country_code: str = "US", 

46 niche: str = "general", 

47) -> dict: 

48 """ 

49 2026 model: tuned for the Shorts-heavy YouTube economy. 

50 

51 All inputs come directly from the YouTube API; no extra worker quota needed. 

52 

53 Args: 

54 total_subs: statistics.subscriberCount 

55 total_views: statistics.viewCount (lifetime) 

56 video_count: statistics.videoCount 

57 country_code: snippet.country (2-letter ISO, e.g. "US") 

58 niche: primary_category from DB (e.g. "Gaming", "Finance") 

59 

60 Returns a dict with keys: 

61 est_monthly_total float — total estimated monthly revenue ($) 

62 revenue_split dict — breakdown: adsense_long, adsense_shorts, brand_deals 

63 assumed_shorts_pct str — e.g. "80%" 

64 """ 

65 rates = _MARKET_DATA.get(country_code.upper(), _MARKET_DATA["DEFAULT"]) 

66 

67 # View-mix heuristic: high video_count relative to subs → Shorts factory 

68 shorts_ratio = 0.8 if (video_count / (total_subs + 1)) > 0.01 else 0.4 

69 

70 # Estimate monthly views (≈6% of lifetime total for an active channel) 

71 monthly_views = total_views * 0.06 

72 views_long = monthly_views * (1 - shorts_ratio) 

73 views_shorts = monthly_views * shorts_ratio 

74 

75 # Niche multiplier — case-insensitive lookup, default 1.0 

76 niche_mod = _NICHE_MULTIPLIERS.get(niche.lower(), 1.0) 

77 

78 # Ad revenue (AdSense) 

79 ad_rev = (views_long / 1_000) * rates["long"] * niche_mod + (views_shorts / 1_000) * rates[ 

80 "shorts" 

81 ] 

82 

83 # Sponsorships: $25 CPM on long-form views only (brands ignore Shorts) 

84 sponsorships = (views_long / 1_000) * 25.0 * niche_mod 

85 

86 # Direct monetisation (merch/memberships): scales with loyal long-form audience 

87 direct_mon = ad_rev * (0.25 if total_subs > 500_000 else 0.05) 

88 

89 total_monthly = ad_rev + sponsorships + direct_mon 

90 

91 return { 

92 "est_monthly_total": round(total_monthly, 2), 

93 "revenue_split": { 

94 "adsense_long": round((views_long / 1_000) * rates["long"] * niche_mod, 2), 

95 "adsense_shorts": round((views_shorts / 1_000) * rates["shorts"], 2), 

96 "brand_deals": round(sponsorships, 2), 

97 }, 

98 "assumed_shorts_pct": f"{int(shorts_ratio * 100)}%", 

99 }

107def format_channel_age(channel_age_days: Optional[int]) -> str: 

108 """ 

109 Format channel age in human-readable format. 

110 

111 Returns: "5y", "8mo", or "New" 

112 """ 

113 if not channel_age_days: 

114 return "New" 

115 

116 years = channel_age_days // 365 

117 if years >= 1: 

118 return f"{years}y" 

119 

120 months = channel_age_days // 30 

121 return f"{months}mo" if months > 0 else "New"

124def get_growth_signal(growth_rate: float) -> Tuple[str, str]: 

125 """ 

126 Get growth signal interpretation. 

127 

128 Returns: 

129 Tuple of (label, style_classes). The label includes the emoji 

130 (e.g. "🚀 Rapid Growth") so callers don't need to handle it separately. 

131 """ 

132 if growth_rate > 5: 

133 return ("🚀 Rapid Growth", "bg-green-100 text-green-800 border-green-300") 

134 elif growth_rate > 1: 

135 return ("📈 Growing", "bg-green-50 text-green-700 border-green-200") 

136 elif growth_rate < -2: 

137 return ("📉 Declining", "bg-red-50 text-red-700 border-red-200") 

138 else: 

139 return ("→ Stable", "bg-gray-50 text-gray-700 border-gray-200")
